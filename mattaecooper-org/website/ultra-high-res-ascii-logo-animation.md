### Particle Physics Parameters
```javascript
// Each particle from the crumbled logo
class Particle {
    constructor(x, y, char, density) {
        this.x = x;
        this.y = y;
        this.char = char;  // The ASCII character (░▒▓█)
        this.vx = (Math.random() - 0.5) * 2;  // Initial horizontal velocity
        this.vy = Math.random() * -2;  // Initial upward velocity
        this.density = density;  // Based on character darkness
        this.mass = density * 0.1;
        this.gravity = 9.81 * this.mass;
        this.airResistance = 0.02;
        this.bounceDecay = 0.3;
        this.settled = false;
    }
    
    update(deltaTime) {
        if (!this.settled) {
            // Apply gravity
            this.vy += this.gravity * deltaTime;
            
            // Apply air resistance
            this.vx *= (1 - this.airResistance);
            this.vy *= (1 - this.airResistance);
            
            // Update position
            this.x += this.vx * deltaTime;
            this.y += this.vy * deltaTime;
            
            // Ground collision
            if (this.y >= GROUND_LEVEL) {
                this.y = GROUND_LEVEL;
                this.vy *= -this.bounceDecay;
                
                // Settle if velocity is low
                if (Math.abs(this.vy) < 0.1) {
                    this.settled = true;
                    this.vy = 0;
                    this.vx = 0;
                }
            }
        }
    }
}
```
